  <style>
    body { font-family: Arial; max-width: 600px; margin: auto; padding: 20px; }
    #chat { border: 1px solid #ccc; padding: 10px; height: 400px; overflow-y: scroll; }
    .user { color: blue; margin: 5px 0; }
    .bot { color: green; margin: 5px 0; }
  </style>

<body>
  <h2>Chatbot Demo</h2>
  <div id="chat"></div>
  <input id="input" type="text" placeholder="Type your message..." style="width:80%;">
  <button onclick="send()">Send</button>

  <script>
    async function send() {
      const text = document.getElementById("input").value;
      document.getElementById("chat").innerHTML += `<div class="user">You: ${text}</div>`;
      document.getElementById("input").value = "";

      const response = await fetch("http://127.0.0.1:8000/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ text })
      });

      const data = await response.json();
      document.getElementById("chat").innerHTML += `<div class="bot">Bot: ${data.response}</div>`;
    }
  </script>
</body>
